import { useEffect, useState } from 'react';
import { useParams } from 'react-router-dom';
import api from '../utils/api';
 
type RecipeDetailsType = {
  title: string;
};
 
type RecipeDetailsProps = {
  strMealThumb: string,
  strMeal: string,
  strCategory: string,
  strInstructions: string,
  strYoutube: string,
};
 
function RecipeDetails({ title }: RecipeDetailsType) {
  const { idDaReceita } = useParams();
  const mealsApi = 'https://www.themealdb.com/api/json/v1/1/lookup.php?i=';
  const drinksApi = 'https://www.thecocktaildb.com/api/json/v1/1/lookup.php?i=';
  const recommendedDrink = 'https://www.thecocktaildb.com/api/json/v1/1/search.php?s=';
  const recommendedMelas = 'https://www.themealdb.com/api/json/v1/1/search.php?s=';
  const [data, setData] = useState<RecipeDetailsProps | null>(null);
  const [recommended, setRecommended] = useState<RecipeDetailsProps | null>(null);
  const [ingredients, setIngredients] = useState<string[]>([]);
  const [meal, setMeal] = useState(false);

  useEffect(() => {
    async function getData() {
      if (title === 'Drinks') {
        const fetchApiDrinks = await api(idDaReceita, drinksApi, 'Filter');
        const fetchRecommendedMeals = await api('', recommendedMelas, 'Filter');
        setData(fetchApiDrinks.drinks[0]);
        setRecommended(fetchRecommendedMeals.meals);
        setIngredients(parseIngredients(fetchApiDrinks.drinks[0]));
      } if (title === 'Meals') {
        setMeal(true);
        const fetchApiMeals = await api(idDaReceita, mealsApi, 'Filter');
        const fetchRecommendedDrinks = await api('', recommendedDrink, 'Filter');
        setData(fetchApiMeals.meals[0]);
        setRecommended(fetchRecommendedDrinks.drinks);
        setIngredients(parseIngredients(fetchApiMeals.meals[0]));
      }
    }
    getData();
  }, [title, drinksApi, mealsApi, idDaReceita]);

  function parseIngredients(dataIng: any | undefined) {
    const dataingredients: string[] = [];
    if (dataIng) {
      for (let i = 1; i <= 20; i++) {
        const ingredient = dataIng[`strIngredient${i}`];
        if (ingredient && ingredient.trim() !== '') {
          const measurement = dataIng[`strMeasure${i}`] || '';
          dataingredients.push(`${measurement} ${ingredient}`);
        }
      }
    }
    return dataingredients;
  }

  function mapRecipeData(recipe: any) {
    return {
      strThumb: recipe.strMealThumb || recipe.strDrinkThumb,
      strName: recipe.strMeal || recipe.strDrink,
      strCategory: recipe.strCategory,
      strAlcoholic: recipe.strAlcoholic,
    };
  }

  if (!data) {
    return <div>Loading...</div>;
  }

  return (
    <div>
      <h1>Recipe Details</h1>
      <br />
      <h1 data-testid="recipe-title">{mapRecipeData(data).strName}</h1>
      <img data-testid="recipe-photo" src={ mapRecipeData(data).strThumb } alt="" />
      {meal ? (
        <h1
          data-testid="recipe-category"
        >
          {mapRecipeData(data).strCategory}
        </h1>
      ) : (
        <h1
          data-testid="recipe-category"
        >
          {mapRecipeData(data).strAlcoholic}
        </h1>
      )}
      <h3>Ingredients:</h3>
      <ul>
        {ingredients.map((ingredient, index) => (
          <li
            data-testid={ `${index}-ingredient-name-and-measure` }
            key={ index }
          >
            { ingredient }
          </li>
        ))}
      </ul>
      <span data-testid="instructions">{data.strInstructions}</span>
      {meal && (
        <span
          data-testid="video"
        >
          {data.strYoutube}
        </span>
      )}
    </div>
  );
}
 
export default RecipeDetails;
